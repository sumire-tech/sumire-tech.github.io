p5.jsでの音楽と連動した波形アニメーション

音声ファイルから。
製作にあたっては、ChatGPTを利用しています。

（音が出ます。最大音量まで大きくしないと聞こえないかと思います）

以下は、p5.jsのコードです。

  //music : Anyone in 2025? しゃろう様

  let sound;
  let fft;
  const numLines = 256;
  let baseRadius = 150;

  let particles = [];
  const numParticles = 200;

  let waveform = [];

  function preload() {
    sound = loadSound('Anyone_in_2025_trim.wav');
  }

  function setup() {
    createCanvas(640, 640);
    angleMode(DEGREES);
    colorMode(HSL);
    fft = new p5.FFT();
    sound.loop();

    for (let i = 0; i < numParticles; i++) {
      particles.push({
        angle: random(360),
        radius: random(baseRadius + 30, baseRadius + 100),
        size: random(1, 3),
        speed: random(0.1, 0.5),
        hue: random(180, 220),
        lightness: random(50, 70),
        flicker: random(0.5, 1.5),
      });
    }
  }

  function draw() {
    background(0, 150);

    translate(width / 2, height / 2);

    if (frameCount % 8 === 0) {
      waveform = fft.waveform(numLines);
    }

    let rotation = frameCount * 0.3;

    strokeWeight(2);
    noFill();

    for (let i = 0; i < numLines; i++) {
      let angle = map(i, 0, numLines, 0, 360) + rotation;

      let gradientPos = (i + frameCount * 0.2) % 60;
      let lightness = 40 + 40 * sin(map(gradientPos, 0, 60, 0, 360));

      let amp = waveform.length > 0 ? waveform[i] : 0;
      let len = baseRadius + amp * 100;

      if (amp > 0.3) {

        drawingContext.shadowBlur = 30;
        drawingContext.shadowColor = 'rgba(255, 255, 255, 0.9)';
        stroke(255, 255, 255);
      } else {

        drawingContext.shadowBlur = 15;
        drawingContext.shadowColor = 'rgba(60, 120, 180, 0.7)';
        stroke(`hsl(200, 100%, ${lightness}%)`);
      }

      let x1 = baseRadius * cos(angle);
      let y1 = baseRadius * sin(angle);
      let x2 = len * cos(angle);
      let y2 = len * sin(angle);

      line(x1, y1, x2, y2);
    }

    drawingContext.shadowBlur = 0;

    colorMode(HSL);
    noStroke();
    for (let p of particles) {
      p.angle += p.speed;
      if (p.angle > 360) p.angle -= 360;

      let waveIndex = floor(map(p.angle, 0, 360, 0, waveform.length - 1));
      let waveVal = waveform.length > 0 ? waveform[waveIndex] : 0;
      let dynamicRadius = p.radius + waveVal * 30 * p.flicker;

      let x = dynamicRadius * cos(p.angle);
      let y = dynamicRadius * sin(p.angle);

      fill(p.hue, 100, p.lightness, 0.7);
      ellipse(x, y, p.size);

      if (random() < 0.03) {
        fill(180, 100, 90, 1);
        ellipse(x, y, p.size * 2);
      }
    }
  }

  function mousePressed() {
    if (!sound.isPlaying()) {
      sound.loop();
    }
  }
